get subscribers for email

SELECT subs.User_ID, subs.Dataset_Name
        FROM tblDataset_Subscriptions as subs
        JOIN tblSubscription_Emails_Linked_Datasets as linkedData
        ON linkedData.Dataset_Name = subs.Dataset_Name
        WHERE lindedData.Email_ID = 1
